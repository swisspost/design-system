# Color

In the following, you can see the Swiss Post specific colors.<br/>
They theme our elements and components.

<PostAlert type="info">
<p slot="heading">Notes for Accessibility</p>
  <p>

  </p>

## Brand colors

<section className="color-swatch__section">

    return <ColorSwatch key={data.key} name={data.key} color={data.value} />;
  })}
</section>

## Signal colors

<section className="color-swatch__section">
  {forEach(SCSS_VARIABLES.signal.solid, function (data) {
    return <ColorSwatch key={data.key} name={data.key} color={data.value} />;
  })}
  {forEach(SCSS_VARIABLES.contextual, function (data) {
    if (data.key === 'info') {
      return <ColorSwatch key={data.key} name={data.key} color={data.value} />;
    }
  })}
  {forEach(SCSS_VARIABLES.signal.background, function (data) {
    return <ColorSwatch key={data.key} name={data.key} color={data.value} />;
  })}
</section>

### Deprecated signal colors

<section className="color-swatch__section">

      return <ColorSwatch key={data.key} name={data.key} color={data.value} />;
    }
  })}
</section>
